### Unchecked - can be changed
<Preview withSource="open">
  <Story name="unchecked - change">
  {() => {
      const [checked, setChecked] = useState(false);
  return (
        <Checkbox 
            id="1" 
            checked={checked} 
            label="Check me!" 
            onChange={() => setChecked(!checked)} 
        />
    )
  }}
  </Story>
</Preview>

### Checked - can be changed
<Preview withSource="open">
  <Story name="checked - change">
  {() => {
      const [checkedByDefault, setCheckedByDefault] = useState(true);
  return (
        <Checkbox
            id="2"
            checked={checkedByDefault}
            label="Checked by default, can be changed"
            onChange={() => setCheckedByDefault(!checkedByDefault)}
        />
    )
  }}
  </Story>
</Preview>
