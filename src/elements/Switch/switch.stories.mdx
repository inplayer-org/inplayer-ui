# Switch


```jsx
import { Switch } from '@inplayer-org/inplayer-ui';

const Page = () => (
    <Switch
        checked={checked}
        id="toggle1"
        label="Checked by default!"
        onChange={() => setChecked(!checked)}
    />
)
```
